#annotating biogeochemical cycling genes in final set of MAGs using DRAM
# Load modules
module purge
module load DRAM/1.3.5-Miniconda3

#change directory
cd /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/
mkdir DRAM
cd DRAM
mkdir DRAM_input_files ##you'll need to move dereplicated MAGs, checkm quality file, and gtdb taxonomy file into DRAM_input_files folder

# Run DRAM: input files are dereplicated MAGs, checkm quality file, and gtdb taxonomy file
DRAM.py annotate -i 'checkm_drep/dedupe_contigs/*.fa' --checkm_quality checkm_drep/dRep.genomeInfo --gtdb_taxonomy gtdbtk_out/gtdbtk.summary.tsv -o Annotations --threads 24

DRAM.py distill -i Annotations/annotations.tsv -o genome_summaries --trna_path Annotations/trnas.tsv --rrna_path Annotations/rrnas.tsv --distillate_gene_names
